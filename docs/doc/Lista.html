def each
   actual = @head
   while actual != nil do
       yield actual.value
       actual = actual.next_
   end
end

def to_s
   string = "("
   dummy = @head
   while (dummy != nil) do
       string += "#{dummy.value.to_s},"
       dummy = dummy.next_
   end
   string += ")"
   return string
end
